} /** --- TRACING ENGINE --- **/ let canvas, ctx, isDrawing = false; function initTracingLogic() { canvas = document.getElementById('trace-canvas-a'); if (!canvas) return; ctx = canvas.getContext('2d'); // Resize canvas to fit wrapper const size = canvas.parentElement.offsetWidth; canvas.width = size; canvas.height = size; // Draw guide letter on back canvas const gCanvas = document.getElementById('guide-canvas-a'); const gCtx = gCanvas.getContext('2d'); gCanvas.width = size; gCanvas.height = size; gCtx.font = `${size * 0.7}px 'Baloo Thambi 2'`; gCtx.fillStyle = '#f0f0f0'; gCtx.textAlign = 'center'; gCtx.textBaseline = 'middle'; gCtx.fillText('அ', size/2, size/2); // Drawing settings ctx.lineWidth = 15; ctx.lineCap = 'round'; ctx.strokeStyle = '#ff9800'; // Events canvas.addEventListener('mousedown', startDraw); canvas.addEventListener('mousemove', draw); canvas.addEventListener('mouseup', stopDraw); canvas.addEventListener('touchstart', (e) => { e.preventDefault(); startDraw(e.touches[0]); }); canvas.addEventListener('touchmove', (e) => { e.preventDefault(); draw(e.touches[0]); }); canvas.addEventListener('touchend', stopDraw); } function startDraw(e) { isDrawing = true; draw(e); } function stopDraw() { isDrawing = false; ctx.beginPath(); calculateAccuracy(); } function draw(e) { if (!isDrawing) return; const rect = canvas.getBoundingClientRect(); const x = (e.clientX || e.pageX) - rect.left; const y = (e.clientY || e.pageY) - rect.top; ctx.lineTo(x, y); ctx.stroke(); ctx.beginPath(); ctx.moveTo(x, y); } function clearCanvas() { ctx.clearRect(0, 0, canvas.width, canvas.height); document.getElementById('acc-bar').style.width = '0%'; document.getElementById('acc-val').innerText = '0'; } function calculateAccuracy() { // Simple pixel-based accuracy simulation const currentAcc = parseInt(document.getElementById('acc-val').innerText); const newAcc = Math.min(currentAcc + 15, 95); // Simulation logic document.getElementById('acc-val').innerText = newAcc; document.getElementById('acc-bar').style.width = newAcc + '%'; if (newAcc > 80) { document.getElementById('result-banner').classList.add('show'); } }
